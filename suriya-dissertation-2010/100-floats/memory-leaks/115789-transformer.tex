public static void jvolveObject(NavigationHistory nh) {
  for (Editor e : nh.editors)
    e.refcount = 0;
  for (History h : nh.history)
    if (h.editor != null)
      h.editor.refcount++;
  for (Editor e : nh.editors)
    if (e.refcount == 0)
      nh.editors.remove(e);
}
